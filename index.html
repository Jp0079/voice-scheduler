    <script>
        // ‚ö†Ô∏è IMPORTANT: Replace this URL with your actual n8n Production Webhook URL
        const WEBHOOK_URL = 'https://jpranav.app.n8n.cloud/webhook/start-scheduling-call';
        
        async function initiateCall() {
            const phone = document.getElementById('phone').value.trim();
            const status = document.getElementById('status');
            const btn = document.getElementById('callBtn');
            
            // Validate phone number
            if (!phone || phone.length < 10) {
                showStatus('‚ö†Ô∏è Please enter a valid phone number with country code', 'error');
                return;
            }
            
            // Disable button
            btn.disabled = true;
            btn.textContent = 'üìû Calling...';
            
            showStatus('Initiating voice call...', 'info');
            
            try {
                const response = await fetch(WEBHOOK_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        phone_number: phone
                    })
                });
                
                const data = await response.json();
                
                if (response.ok && data.success) {
                    showStatus('‚úÖ Call initiated! Please answer your phone. The AI assistant will guide you through scheduling.', 'success');
                    // Keep button disabled
                    setTimeout(() => {
                        btn.disabled = false;
                        btn.textContent = 'üéôÔ∏è Start Voice Call';
                    }, 5000);
                } else {
                    showStatus('‚ùå Failed to initiate call. Please try again.', 'error');
                    btn.disabled = false;
                    btn.textContent = 'üéôÔ∏è Start Voice Call';
                }
            } catch (error) {
                showStatus('‚ùå Error: ' + error.message + '. Please check your connection.', 'error');
                btn.disabled = false;
                btn.textContent = 'üéôÔ∏è Start Voice Call';
            }
        }
        
        function showStatus(message, type) {
            const status = document.getElementById('status');
            status.textContent = message;
            status.className = 'status ' + type;
            status.style.display = 'block';
        }
        
        // Allow Enter key to submit
        document.getElementById('phone').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                initiateCall();
            }
        });
    </script>
